Ejemplos:
#Crea una partición primaria llamada Particion1 de 300 kb
#con el peor ajuste en el disco Disco1.dsk
fdisk -size=300 -PATH=/home/oscar/Documentos/USAC/[ARCHIVOS]/MIA_P1_201603028/CD_Creado/Disco1.dsk -name=Particion1

#Crea una partición extendida dentro de Disco2 de 300 kb
#Tiene el peor ajuste
fdisk -type=E -path=/home/oscar/Documentos/USAC/[ARCHIVOS]/MIA_P1_201603028/CD_Creado/Disco2.dsk -unit=K -name=Particion2 -size=300

#Crea una partición lógica con el mejor ajuste, llamada Partición 3,
#de 1 Mb en el Disco3
fdisk -size=1 -type=L -unit=M -fit=bf -path="/home/oscar/Documentos/USAC/[ARCHIVOS]/MIA_P1_201603028/CDCD Creado/Disco3.dsk"
-name="Particion3"

#Intenta crear una partición extendida dentro de Disco2 de 200 kb
#Debería mostrar error ya que ya existe una partición extendida
#dentro de Disco2
fdisk -type=E -path=/home/oscar/Documentos/USAC/[ARCHIVOS]/MIA_P1_201603028/CD_Creado/Disco2.dsk -name=Part3 -unit=K -size=200

#Elimina de forma rápida una partición llamada Partición 1
fdisk -delete=full -name="Particion1" -PATH=/home/oscar/Documentos/USAC/[ARCHIVOS]/MIA_P1_201603028/CD_Creado/Disco1.dsk

#Elimina de forma completa una partición llamada Partición 1
fdisk -name=Particion1 -delete=full -PATH=/home/oscar/Documentos/USAC/[ARCHIVOS]/MIA_P1_201603028/CD_Creado/Disco1.dsk

#Quitan 500 Kb de Partición 4 en Disco4.dsk
#Ignora los demás parámetros (s)
#Se toma como válido el primero que aparezca, en este caso add
fdisk -add=-500 -size=10 -unit=K -path="/home/oscar/Documentos/USAC/[ARCHIVOS]/MIA_P1_201603028/CDCD Creado/Disco4.dsk"
-name=”Particion4”

#Agrega 1 Mb a la partición Partición 4 del Disco4.dsk
#Se debe validar que haya espacio libre después de la partición
fdisk -add=1 -unit=M -path="/home/oscar/Documentos/USAC/[ARCHIVOS]/MIA_P1_201603028/CDCD Creado/Disco4.dsk" -name="Particion4"
